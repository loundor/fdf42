	image_set_pixel(mlx->image, (int)p1->x, (int)p1->y, clerp(p1->color,
				p2->color, percent));


void	image_set_pixel(t_image *image, int x, int y, int color)
{
	if (x < 0 || x >= WIN_WIDTH || y < 0 || y >= WIN_HEIGHT)
		return ;
	*(int *)(image->ptr + ((x + y * WIN_WIDTH) * image->bpp)) = color;
}
